caretaker_add_hub = {
    owned_planets_only = yes
    sound = event_administrative_work
    resources = {
        category = decisions
        cost = {
            unity = 50
            minerals = 500
        }
    }
    potential = {
        is_homeworld = no
        caretaker_presence_on_planet = no
        exists = owner
        owner = {
            has_origin = origin_caretaker
            NOT = {
                has_country_flag = caretaker_break_up
            }
        }
    }
    allow = {
        always = yes
    }
    effect = {
        add_deposit = d_caretaker_hub
        hidden_effect = {
            set_planet_flag = caretaker_planet_hub_has_been_built
            if = {
                limit = {
                    NOT = {
                        has_planet_flag = caretaker_origin_reminded_to_build_caretaker_hub
                    }
                }
                owner = {
                    change_variable = {
                        which = caretaker_mood
                        value = 2
                    }
                }
            }
        }
    }
    ai_weight = {
        weight = 0
    }
}

                    1 = {
                        set_timed_planet_flag = {
                            flag = caretaker_pop_creation_cooldown
                            days = 720
                        }
                    }
                    2 = {
                        set_timed_planet_flag = {
                            flag = caretaker_pop_creation_cooldown

                        }

            }
            else = {

                    1 = {
                        set_timed_planet_flag = {
                            flag = caretaker_pop_creation_cooldown

                        }
                    }
