        }
        every_owned_leader = {

                }

            if = {
                limit = {

                        leader_class = ruler
                    }
                }
                remove_trait = leader_trait_personal_caretaker
            }
            else = {
                remove_trait = leader_trait_ruler_firstmonitor
            }
        }
        every_pool_leader = {
            limit = {
